  if (!node) return;
  helper(node.left, res);
  res.push(node.val);

var inorderTraversal = function(root) {
  if (!root) return [];
  const res = [];
  const stack = [root];
  let p = root.left;
  while (stack.length !== 0 || p) {
    while (p) {
      stack.push(p);
      p = p.left;
    }

    p = stack.pop();
    res.push(p.val);

    p = p.right;
  }
  return res;
};
